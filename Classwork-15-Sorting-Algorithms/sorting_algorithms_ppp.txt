FUNCTION BubbleSort(numbers):
    GET length of numbers AS n
    FOR sweep FROM 0 TO n - 1 DO:
        FOR pair FROM 0 TO (n - 1 - sweep) DO:
            IF numbers[pair] > numbers[pair + 1] THEN:
                SWAP numbers[pair] AND numbers[pair + 1]
            END IF
        END FOR
    END FOR
END FUNCTION

FUNCTION SelectionSort(numbers):
    GET length of numbers AS n
    FOR i FROM 0 TO n - 1 DO:
        SET min_idx TO i
        FOR j FROM i + 1 TO n - 1 DO:
            IF numbers[j] < numbers[min_idx] THEN:
                SET min_idx TO j
            END IF
        END FOR
        IF min_idx IS NOT EQUAL TO i THEN:
            SWAP numbers[i] AND numbers[min_idx]
        END IF
    END FOR
END FUNCTION

FUNCTION InsertionSort(numbers):
    GET length of numbers AS n
    FOR i FROM 1 TO n - 1 DO:
        SET j TO i
        WHILE j > 0 AND numbers[j - 1] > numbers[j] DO:
            SWAP numbers[j] AND numbers[j - 1]
            SET j TO j - 1
        END WHILE
    END FOR
END FUNCTION

FUNCTION DrawBars(numbers, selected):
    CLEAR canvas
    GET length of numbers AS n
    SET bar_width TO 10.0 / n
    FOR EACH index, number IN numbers DO:
        SET x TO (index * bar_width) + (bar_width / 2)
        IF index IS IN selected THEN:
            SET color TO Red
        ELSE:
            SET color TO Blue
        END IF
        DRAW rectangle at position x representing 'number' with 'color'
    END FOR
    PAUSE canvas for 500 milliseconds
END FUNCTION

FUNCTION BubbleSortAnimated(numbers):
    SET Canvas X scale from -0.1 to 10
    SET Canvas Y scale from -0.5 to (maximum value in numbers + 1)
    GET length of numbers AS n
    FOR sweep FROM 0 TO n - 1 DO:
        FOR pair FROM 0 TO (n - 1 - sweep) DO:
            CALL DrawBars(numbers, selected = pair and pair+1)
            IF numbers[pair] > numbers[pair + 1] THEN:
                SWAP numbers[pair] AND numbers[pair + 1]
                CALL DrawBars(numbers, selected = pair and pair+1)
            END IF
        END FOR
    END FOR
    CALL DrawBars(numbers, selected = none)
    PAUSE canvas for 2000 milliseconds
END FUNCTION

FUNCTION SelectionSortAnimated(numbers):
    SET Canvas X scale from -0.1 to 10
    SET Canvas Y scale from -0.5 to (maximum value in numbers + 1)
    GET length of numbers AS n
    FOR i FROM 0 TO n - 1 DO:
        SET min_idx TO i
        FOR j FROM i + 1 TO n - 1 DO:
            CALL DrawBars(numbers, selected = min_idx and j)
            IF numbers[j] < numbers[min_idx] THEN:
                SET min_idx TO j
            END IF
        END FOR
        IF min_idx IS NOT EQUAL TO i THEN:
            SWAP numbers[i] AND numbers[min_idx]
            CALL DrawBars(numbers, selected = i and min_idx)
        END IF
    END FOR
    CALL DrawBars(numbers, selected = none)
    PAUSE canvas for 2000 milliseconds
END FUNCTION

FUNCTION InsertionSortAnimated(numbers):
    SET Canvas X scale from -0.1 to 10
    SET Canvas Y scale from -0.5 to (maximum value in numbers + 1)
    GET length of numbers AS n
    FOR i FROM 1 TO n - 1 DO:
        SET j TO i
        WHILE j > 0 AND numbers[j - 1] > numbers[j] DO:
            CALL DrawBars(numbers, selected = j and j-1)
            SWAP numbers[j] AND numbers[j - 1]
            CALL DrawBars(numbers, selected = j and j-1)
            SET j TO j - 1
        END WHILE
    END FOR
    CALL DrawBars(numbers, selected = none)
    PAUSE canvas for 2000 milliseconds
END FUNCTION

MAIN PROGRAM:
    START
        GET 10 random numbers between 0 and 100
        SET numbers TO generated list
        PRINT "Before sort: " + numbers
        
        CALL BubbleSortAnimated(numbers)
        
        PRINT "After sort: " + numbers
        PAUSE canvas indefinitely
    END
    END
